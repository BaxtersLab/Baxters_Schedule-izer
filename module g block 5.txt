FILE: app/src/main/java/com/baxter/schedulaizer/ui/transfer/TransferViewModel.kt
Role: State for the Transfer tab.

  Class: TransferViewModel(application: Application) : AndroidViewModel(application)

  private val attachRepo = SchedulaizerApp.get(application).attachmentRepository

  State:
    val pendingUploads: StateFlow<List<AttachmentEntity>>
      = attachRepo.pendingUploads.stateIn(viewModelScope, SharingStarted.Lazily, emptyList())

    val pendingBudgetBlasterSends: StateFlow<List<AttachmentEntity>>
      = attachRepo.pendingBudgetBlasterSends
          .stateIn(viewModelScope, SharingStarted.Lazily, emptyList())

    val tunnelState: StateFlow<String>
      Polls TransportBindings.rd_get_tunnel_state() every 2 seconds.
      Implemented as:
        flow {
          while (true) {
            emit(TransportBindings.rd_get_tunnel_state())
            delay(2_000)
          }
        }.stateIn(viewModelScope, SharingStarted.Lazily, "{}")

    val isConnected: StateFlow<Boolean>
      = flow {
          while (true) {
            emit(TransportBindings.isConnected())
            delay(2_000)
          }
        }.stateIn(viewModelScope, SharingStarted.Lazily, false)

  Functions:
    fun getHomelabDestDir(context: Context): String
      = FileTransferManager.getDestDir(context)

    fun getEndpoint(context: Context): String
      Reads PREF_BUDGET_BLASTER_ENDPOINT from SharedPreferences.
      Returns the configured endpoint string.

    suspend fun retryAll(context: Context)
      For each item in pendingUploads: the upload queue is automatic.
      This function just ensures SchedulaizerService is running so the
      observer loop resumes.
      Starts SchedulaizerService if not running (same logic as MainActivity).

FILE: app/src/main/java/com/baxter/schedulaizer/ui/agent/AgentViewModel.kt
Role: State for the Voice Agent tab.

  Class: AgentViewModel(application: Application) : AndroidViewModel(application)

  private val dispatcher = AgentDispatcher(
      SchedulaizerApp.get(application).eventRepository,
      SchedulaizerApp.get(application).billRepository,
      SchedulaizerApp.get(application).attachmentRepository,
      AlertScheduler,
      application
  )
  private var tts: TextToSpeech? = null

  State:
    private val _transcript = MutableStateFlow<List<Pair<String, String>>>(emptyList())
    val transcript: StateFlow<List<Pair<String, String>>> = _transcript.asStateFlow()
      Each Pair: first = user utterance, second = agent response.

    val voiceStatus: StateFlow<VoiceState.Status> = VoiceState.status

  Functions:

    fun initTts(context: Context)
      Initializes TextToSpeech:
        tts = TextToSpeech(context) { status ->
            if (status == TextToSpeech.SUCCESS) {
                tts?.language = Locale.getDefault()
            }
        }

    fun onVoiceResult(utterance: String)
      Launches viewModelScope.launch(Dispatchers.IO):
        VoiceState.setProcessing()
        val intent = IntentParser.parse(utterance)
        val response = dispatcher.dispatch(intent)
        val updated = _transcript.value + Pair(utterance, response)
        _transcript.value = updated
        VoiceState.setIdle()
        // Speak response on main thread:
        withContext(Dispatchers.Main) { speak(response) }

    fun speak(text: String)
      Calls tts?.speak(text, TextToSpeech.QUEUE_FLUSH, null, null).

    fun clearTranscript()
      _transcript.value = emptyList()

    override fun onCleared()
      tts?.shutdown()
      super.onCleared()


FILE: app/src/main/java/com/baxter/schedulaizer/ui/agent/AgentFragment.kt
Role: Voice agent chat UI.

  Class: AgentFragment : Fragment(R.layout.fragment_agent)

  Private:
    private var broadcastReceiver: BroadcastReceiver? = null

  UI components (FragmentAgentBinding):
    recyclerTranscript: RecyclerView   — scrolling transcript (newest at bottom)
    btnMic: FloatingActionButton       — large mic button (hold/tap to talk)
    textVoiceStatus: TextView          — "Listening..." / "Processing..." / "Ready"
    btnClearTranscript: ImageButton    — clears the transcript list

  Lifecycle:

    onViewCreated():
      viewModel.initTts(requireContext()).
      RecyclerView with TranscriptAdapter (vertical, newest at bottom).
      Observes viewModel.transcript -> submits to adapter, scrolls to last item.
      Observes viewModel.voiceStatus -> updates textVoiceStatus and btnMic icon/tint.
      btnMic click:
        If status == IDLE:
          Check RECORD_AUDIO permission.
          If granted: start VoiceListenerService (ACTION_START_LISTENING).
          If not granted: show rationale Snackbar.
        If status == LISTENING:
          Stop VoiceListenerService (ACTION_STOP_LISTENING).
      btnClearTranscript -> viewModel.clearTranscript().

    onStart():
      Registers broadcastReceiver for ACTION_VOICE_RESULT:
        On receive: reads EXTRA_RECOGNIZED_TEXT.
          If error extra present: VoiceState.setError(message).
          Else: viewModel.onVoiceResult(text).
        LocalBroadcastManager.getInstance().registerReceiver().

    onStop():
      LocalBroadcastManager.getInstance().unregisterReceiver(broadcastReceiver).
      Stops listening: Intent(ACTION_STOP_LISTENING) -> startService().

  TranscriptAdapter (inner class):
    Displays Pair<String,String> — user utterance on right (blue bubble),
    agent response on left (grey bubble).
    Uses two ViewHolder types: USER_BUBBLE and AGENT_BUBBLE.
    Layouts: item_transcript_user.xml, item_transcript_agent.xml.

--------------------------------------------------------------------------------
DATA FLOW
--------------------------------------------------------------------------------
Voice agent full pipeline:
  User taps btnMic -> VoiceListenerService starts
  -> User speaks -> SpeechRecognizer recognizes
  -> ACTION_VOICE_RESULT broadcast -> AgentFragment BroadcastReceiver
  -> viewModel.onVoiceResult(text) -> IntentParser.parse() -> AgentDispatcher.dispatch()
  -> response String -> _transcript updated -> TranscriptAdapter updates
  -> tts.speak(response) -> user hears the response

Transfer tab refresh:
  Every 2 seconds: TransferViewModel polls TransportBindings.isConnected()
  -> textTunnelStatus shows green/red
  -> pendingUploads Flow from Room updates automatically on DB changes

--------------------------------------------------------------------------------
DESIGN DECISIONS
--------------------------------------------------------------------------------
- TextToSpeech is initialized in AgentViewModel (not Fragment) so the TTS engine
  persists across configuration changes (screen rotation). The engine is shutdown
  in onCleared().
- BroadcastReceiver is registered in onStart/onStop (not onResume/onPause) to
  stay active when the user is on the Agent tab but the screen briefly turns off.
- Transcript is displayed with user bubbles right-aligned, agent bubbles
  left-aligned — familiar chat UI pattern requires no user documentation.
- tunnelState polling at 2-second intervals is lightweight (one JNI call) and
  provides responsive feedback without draining battery.

--------------------------------------------------------------------------------
SAFETY & SECURITY NOTES
--------------------------------------------------------------------------------
- RECORD_AUDIO permission is checked in AgentFragment before starting the
  VoiceListenerService on every mic tap. Permission can be revoked between
  taps without crashing.
- BroadcastReceiver is unregistered in onStop() — no audio results are
  processed after the user leaves the tab.
- TextToSpeech speaks the agent's response string. This string is generated
  by AgentDispatcher from controlled templates, not from unsanitized user input
  or database content — TTS injection is not a concern.
